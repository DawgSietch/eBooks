<script>
  let slides;
  let current = 0;
  let isAnimating = false;
  let isFlipbookAnimating = false;

  const animEndEventNames = {
    'WebkitAnimation' : 'webkitAnimationEnd',
    'OAnimation' : 'oAnimationEnd',
    'msAnimation' : 'MSAnimationEnd',
    'animation' : 'animationend'
  };
  const animEndEventName = animEndEventNames[ (function() {
    for (const name in animEndEventNames) {
      if (typeof document.documentElement.style[name] !== 'undefined') {
        return name;
      }
    }
    return '';
  })() ];

  function updatePageCounter() {
    const counter = document.getElementById('page-counter');
    if (counter && slides && slides.length > 0) {
      counter.textContent = `${current + 1} / ${slides.length}`;
    } else if (counter) {
      counter.textContent = "0 / 0";
    }
  }

  function updateNavLinkHighlight() {
    const navLinks = document.querySelectorAll('.top-nav a[data-nav-section]');
    navLinks.forEach(link => link.classList.remove('nav-link-active'));

    let activeSection = '';

    if (current >= 2 && current <= 5) {
      activeSection = 'payments-industry';
    } else if (current >= 6 && current <= 14) {
      activeSection = 'one-click-stay';
    } else if (current >= 15 && current <= 17) {
      activeSection = 'how-we-achieve-success';
    } else if (current >= 18 && current <= 19) {
      activeSection = 'talk-to-us';
    }

    if (activeSection) {
      const activeLink = document.querySelector(`.top-nav a[data-nav-section="${activeSection}"]`);
      if (activeLink) {
        activeLink.classList.add('nav-link-active');
      }
    }
  }

  function showSlide(index, direction = 'next') {
    if (isAnimating || index === current || !slides || slides.length === 0) {
      return;
    }

    isAnimating = true;

    const oldSlide = slides[current];
    const newSlide = slides[index];

    if (!oldSlide || !newSlide) {
      console.error("Slide element not found during transition. Old Slide:", oldSlide, "New Slide:", newSlide);
      isAnimating = false;
      return;
    }

    newSlide.style.visibility = 'visible';
    newSlide.classList.add('pt-page-ontop');
    newSlide.classList.add('pt-page-current');

    if (direction === 'next') {
      newSlide.classList.add('pt-page-moveFromRight');
      oldSlide.classList.add('pt-page-exit-left');
    } else {
      newSlide.classList.add('pt-page-moveFromLeft');
      oldSlide.classList.add('pt-page-exit-right');
    }

    oldSlide.classList.remove('pt-page-current');

    newSlide.addEventListener(animEndEventName, function handleNewSlideAnimationEnd() {
      newSlide.removeEventListener(animEndEventName, handleNewSlideAnimationEnd);

      oldSlide.classList.remove('pt-page-exit-left', 'pt-page-exit-right');
      oldSlide.style.visibility = 'hidden';

      newSlide.classList.remove('pt-page-moveFromRight', 'pt-page-moveFromLeft', 'pt-page-ontop');

      isAnimating = false;
    });

    current = index;
    updatePageCounter();
    updateNavLinkHighlight();
  }

  function nextArrow() {
    let nextIndex = current + 1;
    if (nextIndex >= slides.length) {
      nextIndex = 0;
    }
    showSlide(nextIndex, 'next');
  }

  function prevArrow() {
    let prevIndex = current - 1;
    if (prevIndex < 0) {
      prevIndex = slides.length - 1;
    }
    showSlide(prevIndex, 'prev');
  }

  function goToSlide(index) {
    const direction = index > current ? 'next' : 'prev';
    showSlide(index, direction);
  }

  function unlockContent() {
    const form = document.getElementById('unlock-form');
    if (form && !form.checkValidity()) {
        form.reportValidity();
        return;
    }

    console.log("unlockContent called");
    document.getElementById('unlock').style.display = 'none';

    const ptPages = document.querySelector('.pt-pages');
    const topNav = document.querySelector('.top-nav');
    const navButtons = document.querySelectorAll('.nav-button');
    const pageCounter = document.getElementById('page-counter');

    console.log("Elements selected:", { ptPages, topNav, navButtons, pageCounter });

    if (ptPages) ptPages.classList.add('active');
    if (topNav) {
      topNav.classList.add('visible');
      console.log("topNav classList after adding visible:", topNav.classList);
    }
    navButtons.forEach(btn => {
      if (btn) btn.classList.add('visible');
    });
    if (pageCounter) pageCounter.classList.add('visible');

    localStorage.setItem('ebookUnlocked', 'true');

    if (slides.length > 0) {
      slides[0].classList.add('pt-page-current');
      slides[0].style.visibility = 'visible';
    } else {
      console.error("No slides found to display after unlock.");
    }
    current = 0;
    updatePageCounter();
    updateNavLinkHighlight();
  }

  let currentFlipbookPage = 1;
  const totalFlipbookPages = 10;

  function showFlipbookPage(pageNumber) {
    if (isFlipbookAnimating) return;
    
    const pageDots = document.querySelectorAll('.flipbook-nav .page-dot');
    
    const activePage = document.querySelector('.flipbook-page-active');
    if (activePage && pageNumber === currentFlipbookPage) {
        return;
    }

    isFlipbookAnimating = true;

    const oldFlipbookPage = document.querySelector('.flipbook-page-active');
    const newFlipbookPage = document.getElementById(`flipbook-page-${pageNumber}`);
    
    // Update dots
    pageDots.forEach(dot => dot.classList.remove('active-dot'));
    if (pageDots[pageNumber - 1]) {
        pageDots[pageNumber - 1].classList.add('active-dot');
    }

    // Fade transition logic
    if (oldFlipbookPage) {
        oldFlipbookPage.style.transition = 'opacity 0.5s ease-in-out';
        oldFlipbookPage.style.opacity = '0';
        oldFlipbookPage.classList.remove('flipbook-page-active');
    }

    newFlipbookPage.style.transition = 'opacity 0.5s ease-in-out';
    newFlipbookPage.style.opacity = '0';
    newFlipbookPage.style.visibility = 'visible';
    newFlipbookPage.style.zIndex = '2';

    setTimeout(() => {
        newFlipbookPage.style.opacity = '1';
        newFlipbookPage.classList.add('flipbook-page-active');
    }, 50);

    setTimeout(() => {
        if (oldFlipbookPage) {
            oldFlipbookPage.style.visibility = 'hidden';
            oldFlipbookPage.style.zIndex = '0';
        }
        newFlipbookPage.style.zIndex = '1';
        currentFlipbookPage = pageNumber;
        isFlipbookAnimating = false;
    }, 550);
  }


  document.addEventListener('DOMContentLoaded', () => {
    slides = document.querySelectorAll('.pt-page');

    if (localStorage.getItem('ebookUnlocked') === 'true') {
        unlockContent();
    }

    window.unlockContent = unlockContent;
    window.nextArrow = nextArrow;
    window.prevArrow = prevArrow;
    window.goToSlide = goToSlide;
    window.showFlipbookPage = showFlipbookPage;
    window.closeBenefitOverlay = closeBenefitOverlay;

    document.addEventListener('keydown', (event) => {
      if (document.getElementById('unlock').style.display === 'none') {
        if (event.key === 'ArrowRight') {
          if (current === 14) {
            let nextFlipbookPage = currentFlipbookPage + 1;
            if (nextFlipbookPage > totalFlipbookPages) {
              nextFlipbookPage = 1;
            }
            showFlipbookPage(nextFlipbookPage);
          } else {
            nextArrow();
          }
        } else if (event.key === 'ArrowLeft') {
          if (current === 14) {
            let prevFlipbookPage = currentFlipbookPage - 1;
            if (prevFlipbookPage < 1) {
              prevFlipbookPage = totalFlipbookPages;
            }
            showFlipbookPage(prevFlipbookPage);
          } else {
            prevArrow();
          }
        }
      }
    });

    const benefitHeaders = document.querySelectorAll('.benefit-header');
    const overlayContent = document.querySelector('.benefit-overlay-content');
    const overlayTitle = document.getElementById('overlay-title');
    const overlayText = document.getElementById('overlay-text');
    const closeButton = document.querySelector('.benefit-overlay-content .close-button');

    benefitHeaders.forEach(header => {
      header.addEventListener('click', () => {
        const benefitItem = header.closest('.benefit-item');
        const title = header.querySelector('.benefit-title').textContent;
        const content = benefitItem.querySelector('.benefit-content-hidden').innerHTML;
        const headerBgColor = window.getComputedStyle(header).backgroundColor;

        overlayTitle.textContent = title;
        overlayText.innerHTML = content;
        overlayContent.style.backgroundColor = headerBgColor;

        overlayContent.classList.add('active');
      });
    });

    function closeBenefitOverlay() {
      overlayContent.classList.remove('active');
    }

    closeButton.addEventListener('click', closeBenefitOverlay);
    
    // Initialize Flipbook
    const flipbookNavContainers = document.querySelectorAll('.flipbook-nav');
    flipbookNavContainers.forEach(nav => {
        nav.innerHTML = ''; // Clear existing dots
        for (let i = 0; i < totalFlipbookPages; i++) {
            const dot = document.createElement('span');
            dot.classList.add('page-dot');
            dot.textContent = i + 1;
            if (i === 0) {
                dot.classList.add('active-dot');
            }
            dot.addEventListener('click', () => showFlipbookPage(i + 1));
            nav.appendChild(dot);
        }
    });
    
    // Show the first page by default without animation
    if (document.querySelector('.flipbook-container')) {
      const firstPage = document.getElementById('flipbook-page-1');
      if (firstPage) {
        firstPage.style.transition = 'none';
        firstPage.style.visibility = 'visible';
        firstPage.style.opacity = '1';
        firstPage.classList.add('flipbook-page-active');
      }
    }


    updatePageCounter();
    updateNavLinkHighlight();
  });
</script>
